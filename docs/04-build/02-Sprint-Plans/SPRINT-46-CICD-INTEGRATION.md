## Week 1: GitHub Action & Pre-commit (Mar 17-21)

### Day 1-2: GitHub Action

**Task**: Create reusable GitHub Action for SDLC validation

```yaml
# .github/actions/sdlc-validate/action.yml

name: 'SDLC Structure Validation'
description: 'Validate SDLC folder structure and naming conventions'
author: 'SDLC Orchestrator'

branding:
  icon: 'check-circle'
  color: 'green'

inputs:
  docs-path:
    description: 'Path to docs folder'
    required: false
    default: 'docs'
  
  config-path:
    description: 'Path to .sdlc-config.json'
    required: false
    default: '.sdlc-config.json'
  
  fail-on-warning:
    description: 'Fail if warnings found'
    required: false
    default: 'false'
  
  auto-fix:
    description: 'Automatically fix violations'
    required: false
    default: 'false'
  
  create-pr:
    description: 'Create PR with fixes (requires auto-fix: true)'
    required: false
    default: 'false'

outputs:
  violations-count:
    description: 'Number of violations found'
  errors-count:
    description: 'Number of errors'
  warnings-count:
    description: 'Number of warnings'
  fix-applied:
    description: 'Whether fixes were applied'

runs:
  using: 'composite'
  steps:
    - name: Setup Python
      uses: actions/setup-python@v5
      with:
        python-version: '3.11'
    
    - name: Install sdlcctl
      shell: bash
      run: |
        pip install sdlcctl
    
    - name: Run validation
      id: validate
      shell: bash
      run: |
        ARGS="--format github"
        
        if [ "${{ inputs.fail-on-warning }}" == "true" ]; then
          ARGS="$ARGS --fail-on-warning"
        fi
        
        if [ "${{ inputs.auto-fix }}" == "true" ]; then
          ARGS="$ARGS --auto-fix"
        fi
        
        sdlcctl validate ${{ inputs.docs-path }} \
          --config ${{ inputs.config-path }} \
          $ARGS | tee validation-output.txt
        
        # Parse output for GitHub annotations
        python << 'EOF'
        import re
        import os
        
        with open('validation-output.txt') as f:
            content = f.read()
        
        # Count violations
        errors = len(re.findall(r'::error', content))
        warnings = len(re.findall(r'::warning', content))
        
        # Set outputs
        with open(os.environ['GITHUB_OUTPUT'], 'a') as f:
            f.write(f'violations-count={errors + warnings}\n')
            f.write(f'errors-count={errors}\n')
            f.write(f'warnings-count={warnings}\n')
        EOF
    
    - name: Create Fix PR
      if: inputs.auto-fix == 'true' && inputs.create-pr == 'true'
      uses: peter-evans/create-pull-request@v5
      with:
        title: '🔧 Auto-fix SDLC structure violations'
        body: |
          This PR was automatically generated by SDLC Orchestrator.
          
          ## Violations Fixed
          See commit message for details.
          
          ## Validation Report
          ```
          ${{ steps.validate.outputs.violations-count }} violations found
          ${{ steps.validate.outputs.errors-count }} errors
          ${{ steps.validate.outputs.warnings-count }} warnings
          ```
        branch: sdlc/auto-fix
        commit-message: 'refactor(sdlc): Auto-fix structure violations'
```

# backend/sdlcctl/commands/validate.py

@app.command()
def validate(
    staged_only: bool = typer.Option(
        False,
        "--staged-only",
        help="Only validate staged files (for pre-commit)"
    ),
):
    """Validate SDLC structure"""
    
    if staged_only:
        # Get staged files from git
        result = subprocess.run(
            ["git", "diff", "--cached", "--name-only", "--diff-filter=ACMR"],
            capture_output=True,
            text=True
        )
        staged_files = [
            Path(f) for f in result.stdout.strip().split("\n")
            if f.startswith("docs/") and f.endswith(".md")
        ]
        
        if not staged_files:
            console.print("[green]No docs files staged[/green]")
            return
        
        console.print(f"[cyan]Validating {len(staged_files)} staged files...[/cyan]")
        
        # Only validate these specific files
        scanner = SDLCStructureScanner(docs_path)
        violations = scanner.scan(paths=staged_files)
    else:
        # Full validation
        violations = scanner.scan()
```

### Day 5: PR Comment Trigger

**Task**: Implement `/sdlc fix` command in PR comments

```python
# backend/app/api/routes/github_webhook.py

from fastapi import APIRouter, Request, HTTPException
from app.services.github_service import GitHubService
from app.services.sdlc_fixer_service import SDLCFixerService

router = APIRouter()

@router.post("/webhook/github")
async def handle_github_webhook(request: Request):
    """Handle GitHub webhooks for /sdlc commands"""
    
    event = request.headers.get("X-GitHub-Event")
    payload = await request.json()
    
    if event == "issue_comment":
        return await handle_comment(payload)
    
    return {"status": "ignored"}

async def handle_comment(payload: dict):
    """Handle PR comment commands"""
    
    comment_body = payload.get("comment", {}).get("body", "")
    
    # Check for /sdlc commands
    if not comment_body.strip().startswith("/sdlc"):
        return {"status": "not_command"}
    
    command = comment_body.strip().split()[1] if len(comment_body.split()) > 1 else None
    
    if command == "fix":
        return await handle_fix_command(payload)
    elif command == "validate":
        return await handle_validate_command(payload)
    else:
        return {"status": "unknown_command", "command": command}

async def handle_fix_command(payload: dict):
    """
    Handle /sdlc fix command
    
    Workflow:
    1. Clone PR branch
    2. Run sdlcctl validate --auto-fix
    3. Push changes
    4. Comment with results
    """
    
    repo = payload["repository"]["full_name"]
    pr_number = payload["issue"]["number"]
    
    github = GitHubService()
    fixer = SDLCFixerService()
    
    # React to comment
    await github.add_reaction(
        repo=repo,
        comment_id=payload["comment"]["id"],
        reaction="eyes"
    )
    
    try:
        # Get PR details
        pr = await github.get_pull_request(repo, pr_number)
        branch = pr["head"]["ref"]
        
        # Clone and fix
        result = await fixer.fix_pr(
            repo=repo,
            branch=branch,
            pr_number=pr_number
        )
        
        # Comment with results
        if result.success:
            comment = f"""
## ✅ SDLC Structure Fixed

**Violations fixed:** {result.fixed_count}
**Commit:** `{result.commit_hash}`

<details>
<summary>Changes made</summary>

```
{result.changes_summary}
```

</details>

_Generated by SDLC Orchestrator_
"""
        else:
            comment = f"""
## ❌ SDLC Fix Failed

**Error:** {result.error_message}

Please run `sdlcctl validate --auto-fix` locally to debug.

_Generated by SDLC Orchestrator_
"""
        
        await github.create_comment(repo, pr_number, comment)
        
        # Update reaction
        await github.add_reaction(
            repo=repo,
            comment_id=payload["comment"]["id"],
            reaction="rocket" if result.success else "confused"
        )
        
        return {"status": "success", "fixed": result.fixed_count}
        
    except Exception as e:
        await github.create_comment(
            repo, pr_number, 
            f"## ❌ Error\n\n```\n{str(e)}\n```"
        )
        return {"status": "error", "message": str(e)}
```

### Day 6-7: Dashboard Integration

**Task**: Add SDLC compliance metrics to web dashboard

```typescript
// frontend/web/src/pages/SDLCCompliancePage.tsx

import React from 'react';
import { useQuery } from '@tanstack/react-query';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Progress } from '@/components/ui/progress';

interface ComplianceMetrics {
  total_files: number;
  compliant_files: number;
  violations: {
    errors: number;
    warnings: number;
    info: number;
  };
  by_rule: Record<string, number>;
  last_scan: string;
}

export function SDLCCompliancePage() {
  const { data: metrics, isLoading } = useQuery<ComplianceMetrics>({
    queryKey: ['sdlc-compliance'],
    queryFn: () => fetch('/api/v1/sdlc/compliance/metrics').then(r => r.json()),
    refetchInterval: 60000, // Refresh every minute
  });

  if (isLoading) return <div>Loading...</div>;
  if (!metrics) return null;

  const complianceRate = (metrics.compliant_files / metrics.total_files) * 100;

  return (
    <div className="container mx-auto py-8">
      <h1 className="text-3xl font-bold mb-8">SDLC Structure Compliance</h1>
      
      {/* Overview Cards */}
      <div className="grid grid-cols-1 md:grid-cols-4 gap-4 mb-8">
        <Card>
          <CardHeader className="pb-2">
            <CardTitle className="text-sm font-medium">Compliance Rate</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-2xl font-bold">{complianceRate.toFixed(1)}%</div>
            <Progress value={complianceRate} className="mt-2" />
          </CardContent>
        </Card>
        
        <Card>
          <CardHeader className="pb-2">
            <CardTitle className="text-sm font-medium">Total Files</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-2xl font-bold">{metrics.total_files}</div>
          </CardContent>
        </Card>
        
        <Card>
          <CardHeader className="pb-2">
            <CardTitle className="text-sm font-medium">Errors</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-2xl font-bold text-red-500">
              {metrics.violations.errors}
            </div>
          </CardContent>
        </Card>
        
        <Card>
          <CardHeader className="pb-2">
            <CardTitle className="text-sm font-medium">Warnings</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-2xl font-bold text-yellow-500">
              {metrics.violations.warnings}
            </div>
          </CardContent>
        </Card>
      </div>
      
      {/* Violations by Rule */}
      <Card className="mb-8">
        <CardHeader>
          <CardTitle>Violations by Rule</CardTitle>
        </CardHeader>
        <CardContent>
          <div className="space-y-4">
            {Object.entries(metrics.by_rule).map(([rule, count]) => (
              <div key={rule} className="flex items-center justify-between">
                <div className="flex items-center gap-2">
                  <Badge variant={count > 0 ? "destructive" : "secondary"}>
                    {rule}
                  </Badge>
                  <span className="text-sm text-muted-foreground">
                    {getRuleDescription(rule)}
                  </span>
                </div>
                <span className="font-mono">{count}</span>
              </div>
            ))}
          </div>
        </CardContent>
      </Card>
      
      {/* Last Scan Info */}
      <p className="text-sm text-muted-foreground">
        Last scanned: {new Date(metrics.last_scan).toLocaleString()}
      </p>
    </div>
  );
}

function getRuleDescription(rule: string): string {
  const descriptions: Record<string, string> = {
    'STAGE-001': 'Invalid stage folder naming',
    'STAGE-002': 'Unknown stage number',
    'STAGE-003': 'Stage name mismatch',
    'NUM-001': 'Duplicate subfolder numbering',
    'NAME-001': 'Non-standard folder name',
    'NAME-002': 'Non-standard file name',
    'HDR-001': 'Missing header table',
    'HDR-002': 'Missing required header field',
  };
  return descriptions[rule] || 'Unknown violation';
}
```

## Troubleshooting

### Common Issues

**Q: Why am I getting false positives?**

A: Check your `.sdlc-config.json` for custom rules. Some projects use non-standard naming that can be whitelisted.

**Q: Auto-fix broke my links!**

A: The cross-reference updater handles most cases. If links are still broken:
1. Run `sdlcctl validate` to check for `XREF-001` violations
2. Manually fix any remaining broken links
3. Submit a bug report with the original structure

**Q: Pre-commit is too slow**

A: Use `--staged-only` to only validate changed files:
```yaml
hooks:
  - id: sdlc-validate
    args: ['--staged-only']
```
